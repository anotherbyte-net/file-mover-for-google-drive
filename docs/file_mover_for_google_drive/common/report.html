 43class ReportCsv:
 44    """A CSV file report."""
 45
 46    def __init__(self, top_dir: pathlib.Path, report: type[TypeReport_co]):
 47        """Create a new csv Report instance.
 48
 49        Args:
 50            top_dir: The directory for the csv files.
 51            report: The type of report to create.
 52        """
 53        self._top_dir = top_dir
 54        self._report = report
 55
 56        self._file_date = datetime.now()
 57        self._writer: typing.Optional[csv.DictWriter] = None
 58        self._file_path: typing.Optional[pathlib.Path] = None
 59        self._file_handle: typing.Optional[typing.TextIO] = None
 60
 61    @property
 62    def path(self):
 63        """Get the file path.
 64
 65        Returns:
 66            The file path.
 67        """
 68        return self._file_path
 69
 70    def start(self) -> None:
 71        """Start the report file.
 72
 73        This sets up the resources to write to the report file.
 74
 75        Returns:
 76            None
 77        """
 78
 79        name = self._report.report_name()
 80        fields = self._report.fields()
 81
 82        file_date_str = self._file_date.isoformat(sep="-", timespec="seconds")
 83        file_date_str = file_date_str.replace(":", "-")
 84
 85        file_name = f"{file_date_str}-{name}"
 86        report_file = (self._top_dir / file_name).with_suffix(".csv")
 87        report_file.parent.mkdir(exist_ok=True, parents=True)
 88
 89        self._file_path = report_file
 90        self._file_handle = open(
 91            report_file,
 92            "wt",
 93            newline="",
 94            encoding="utf-8",
 95        )  # noqa: R1732
 96        self._writer = csv.DictWriter(self._file_handle, fields)
 97        self._writer.writeheader()
 98
 99    def write_item(self, item: typing.Mapping) -> None:
100        """Write an item to the report file.
101
102        Args:
103            item: The item to write.
104
105        Returns:
106            None
107        """
108        if self._writer:
109            self._writer.writerow(item)
110        else:
111            raise ValueError("Csv writer is not ready.")
112
113    def read(self, name: str) -> typing.Iterable[TypeReport_co]:
114        """Read the report file.
115
116        Args:
117            name: The name of the report file to read.
118
119        Returns:
120            The contents of the report file.
121        """
122        report_file = (self._top_dir / name).with_suffix(".csv")
123        if not report_file.exists():
124            raise ValueError(f"Plan file does not exist '{report_file}'.")
125        with open(report_file, "rt", encoding="utf-8") as handle:
126            reader = csv.DictReader(handle)
127            for row in reader:
128                yield self._report(**row)
129
130    def finish(self) -> None:
131        """Finish the report and close the file.
132
133        This ensures the resources to write the file are closed.
134
135        Returns:
136            None
137        """
138
139        if self._file_handle:
140            self._file_handle.close()
141            self._file_handle = None
142
143        if self._writer:
144            self._writer = None
