206    def execute_batch(
207        self,
208        requests: list[http.HttpRequest],
209        callback: typing.Callable[
210            [str, typing.Any, typing.Optional[http.HttpError]], None
211        ],
212    ) -> None:
213        """Execute a batch of operations.
214
215        Args:
216            requests: The requests to execute as a batch.
217            callback: The callback for the result of each request.
218
219        Returns:
220            None
221        """
222
223        # https://developers.google.com/drive/api/guides/performance#batch-requests
224        # https://developers.google.com/drive/api/guides/manage-sharing#change-multiple-permissions
225
226        # Batch requests with more than 100 calls may result in an error.
227        # There is an 8000-character limit on the length
228        # of the URL for each inner request.
229        # Currently, Google Drive does not support batch operations for media,
230        # either for upload or download.
231        operations_per_batch = 100
232        for index, operations_group in enumerate(
233            self._batched(requests, operations_per_batch)
234        ):
235            batch = self.client.new_batch_http_request(callback=callback)
236            for operation in operations_group:
237                if operation:
238                    batch.add(operation)
239
240            try:
241                logger.debug(
242                    "Sending batch %s with %s operations.", index + 1, len(batch)
243                )
244                self._do_execute(batch)
245                logger.debug("Sent batch %s.", index + 1)
246            except http.HttpError as error:
247                logger.error(
248                    "An error occurred in batch %s: %s", index + 1, error, exc_info=True
249                )

289    def _batched(
290        self,
291        iterable: typing.Iterable[http.HttpRequest],
292        count: int = 1,
293        fill_value: typing.Any = None,
294    ) -> typing.Iterable[typing.Iterable[http.HttpRequest]]:
295        """Collect data into fixed-length chunks or blocks.
296
297        Args:
298            iterable: The iterables to break into batches.
299            count: The number of items per batch.
300            fill_value: The value to fill shorter batches.
301
302        Returns:
303            An iterable of batches.
304        """
305
306        # e.g. _batched('ABCDEFG', 3, 'x') --> ABC DEF Gxx"
307        # https://docs.python.org/3.5/library/itertools.html#itertools-recipes
308        args = [iter(iterable)] * count
309        return itertools.zip_longest(*args, fillvalue=fill_value)

489    def delete_permissions(
490        self, entry: models.GoogleDriveEntry
491    ) -> list[http.HttpRequest]:
492        """Delete permissions for entry that are not the owner or current user."""
493
494        permission_user = models.GoogleDrivePermissionTypeOptions.USER
495        permission_anyone = models.GoogleDrivePermissionTypeOptions.ANYONE
496        role_owner = models.GoogleDrivePermissionRoleOptions.OWNER.value
497
498        operations = []
499        for permission in entry.permissions_all:
500            is_owner = permission.role == role_owner
501            is_anyone = permission.entry_type == permission_anyone
502            is_user = permission.entry_type == permission_user
503            is_current_user = permission.user_email == self._account.account_id
504
505            if is_owner or (is_user and is_current_user):
506                logger.debug('Keep permission: "%s".', str(permission))
507
508            elif is_anyone or (is_user and not is_current_user):
509                operation = self.delete_permission(entry.entry_id, permission.entry_id)
510                operations.append(operation)
511
512            else:
513                raise ValueError(f'Don\'t know about permission "{str(permission)}".')
514

603class GoogleDriveActions:
604    """
605    Interacts with a particular Google 'My Drive' or 'Shared Drive'
606    by executing requests that have been built.
607    """
608
609    def __init__(self, container: GoogleDriveContainer):

616    def get_descendants(
617        self, folder_id: str
618    ) -> typing.Generator[models.GoogleDriveEntry, typing.Any, None]:
619        """
620        Get all descendants of the given folder.
621        In other words, get the files in the folder,
622        then the files in each sub-folder, recursively.
623        """
624
625        container = self._container
626        api = container.api
627
628        request = container.get_children(folder_id)
629        for entry_data in api.execute_files_list(request):
630            # provide the children of folder_id
631            entry = self._get_entry(entry_data)
632            yield entry
633
634            if entry.entry_id != folder_id and entry.is_dir:
635                # provide the children of folders in folder_id
636                for entry_desc in self.get_descendants(entry.entry_id):
637                    yield entry_desc
638
639    def get_entry(self, entry_id: str) -> models.GoogleDriveEntry:
640        """Get the details of a file or folder."""
641
642        container = self.container
643        api = container.api
644
645        request = container.get_entry(entry_id)
646        entry_data = api.execute_single(request)
647        entry = self._get_entry(entry_data)
648
649        return entry

651    def get_pair_copy_entry(
652        self, entry: models.GoogleDriveEntry
653    ) -> typing.Optional[models.GoogleDriveEntry]:
654        """
655        For an original (un-owned), get a copy if it exists.
656        For a copy (owned) file, get the original.
657
658        Works in personal accounts only.
659        """
660
661        container = self._container
662        api = container.api
663        config = api.config
664        account = config.account
665
666        account_type_personal = models.GoogleDriveAccountTypeOptions.PERSONAL
667        if account.account_type != account_type_personal:
668            raise ValueError(
669                f"Entry pairs only work in '{account_type_personal}' accounts, "
670                f"not '{account.account_type}'."
671            )
672
673        account_id = account.account_id
674        entry_is_owned = entry.is_owned_by(account_id)
675
676        key_copy = models.GoogleDrivePropertyKeyOptions.CUSTOM_COPY_FILE_ID.value
677        key_orig = models.GoogleDrivePropertyKeyOptions.CUSTOM_ORIGINAL_FILE_ID.value
678
679        if entry_is_owned:
680            # if the entry is owned (might be a copy),
681            # see if there is an original that is not owned
682            prop_key = key_copy
683
684            # check the found original (unowned) entry's property
685            pair_key = key_orig
686
687        else:
688            # if the entry is not owned (the original),
689            # see if there is a copy that is owned
690            prop_key = key_orig
691
692            # check the found copy (owned) entry's property
693            pair_key = key_copy
694
695        if not prop_key:
696            return None
697
698        # get the pair
699        prop_value = entry.entry_id
700        request = container.get_entries_by_property(prop_key, prop_value)
701
702        entries = []
703        for entry_data in api.execute_files_list(request):
704            entry_item = self._get_entry(entry_data)
705            entries.append(entry_item)
706
707        entry_count = len(entries)
708        if entry_count > 1:
709            raise ValueError(
710                f"More than one match for property '{prop_key}={prop_value}'."
711            )
712
713        if entry_count < 1:
714            return None
715
716        # found the pair
717        pair_entry = entries[0]
718
719        # check that the pair has the expected id stored in the entry
720        expected_pair_id = entry.properties_shared.get(pair_key)
721        actual_pair_id = pair_entry.entry_id
722
723        if expected_pair_id != actual_pair_id:
724            raise ValueError(
