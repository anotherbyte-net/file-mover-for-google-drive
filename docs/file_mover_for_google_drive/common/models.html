386    e.g. '11823143700967846661', 'anyoneWithLink'

389    """The email address of the user or group to which this permission refers."""
390    domain: typing.Optional[str] = None
391    """The domain to which this permission refers.
392
393    - The entire domain, such as "your-company.com."
394    - A target audience, such as "ID.audience.googledomains.com."
395    """
396    display_name: typing.Optional[str] = None
397    """The "pretty" name of the value of the permission.
398        The following is a list of examples for each type of permission:
399
400        - user - User's full name.
401        - group - Name of the Google Group, such as 'The Company Administrators'
402        - domain - String domain name, such as 'your-company.com'
403        - anyone - No displayName is present.
404
405        Returns:
406            The display name.
407        """

409    @classmethod
410    def load_data(cls, data: typing.Mapping) -> "GoogleDrivePermission":
411        params = {}
412
413        entry_id = data.get("id")
414        if not entry_id or not entry_id.strip():
415            raise ValueError("Permission must include 'id'.")
416        params["entry_id"] = entry_id
417
418        entry_type = data.get("type")
419        if not entry_type:
420            raise ValueError("Permission must include 'type'.")
421        entry_type = GoogleDrivePermissionTypeOptions(entry_type)
422        params["entry_type"] = entry_type
423
424        role = data.get("role")
425        if not role:
426            raise ValueError("Permission must include 'role'.")
427        params["role"] = GoogleDrivePermissionRoleOptions(role)
428
429        if entry_type in [
430            GoogleDrivePermissionTypeOptions.USER,
431            GoogleDrivePermissionTypeOptions.GROUP,
432        ]:
433            params["user_email"] = data.get("emailAddress")
434
435        display_name = data.get("displayName")
436        if display_name:
437            params["display_name"] = str(display_name)
438
439        if entry_type == GoogleDrivePermissionTypeOptions.DOMAIN:
440            params["domain"] = data.get("domain")
441
442        result = GoogleDrivePermission(**params)
443
444        if not result.entry_type:
445            raise ValueError("Entry type is not set.")
446        if not result.role:
447            raise ValueError("Role is not set.")
448        if not result.entry_id:
449            raise ValueError("Entry id is not set.")
450        if (
451            not result.display_name
452            and result.entry_type != GoogleDrivePermissionTypeOptions.ANYONE
453        ):
454            raise ValueError("Display name is not set.")
455        if not result.user_email and result.entry_type in [
456            GoogleDrivePermissionTypeOptions.USER,
457            GoogleDrivePermissionTypeOptions.GROUP,
458        ]:
459            raise ValueError("User email must be set for type 'user' or 'group'.")
460        if not result.domain and result.entry_type in [
461            GoogleDrivePermissionTypeOptions.DOMAIN,
462        ]:
463            raise ValueError("Domain must be set for type 'domain'.")
464
465        return result

583        if "fileMoverExtraAccount" not in data:
584            raise ValueError("Must include the account information.")
585        if "fileMoverExtraPermissions" not in data:
586            raise ValueError("Must include the response from permissions.list.")
587

589            GoogleDrivePermission.load_data(i) for i in data.get("permissions", [])
590        ]

592        params = {
593            "permissions_all": included_permissions,
594            "account": data.get("fileMoverExtraAccount"),
595        }
596
597        entry_id = data.get("id")
598        if not entry_id or not entry_id.strip():
599            raise ValueError("Entry must include 'id'.")
600        params["entry_id"] = entry_id
601
602        ids: list[str] = data.get("parents", [])
603        if len(ids) != 1:

605        params["parent_id"] = ids[0]

611        date_modified = data.get("modifiedTime")
612        if date_modified:
613            params["date_modified"] = datetime.fromisoformat(date_modified)
614
615        params["name"] = data.get("name")
616        params["description"] = data.get("description")
617        params["mime_type"] = data.get("mimeType")
618        params["view_link"] = data.get("webViewLink")
619        params["size_bytes"] = int(str(data.get("size", "0")))
620        params["quota_bytes"] = int(str(data.get("quotaBytesUsed", "0")))
621        params["checksum_sha256"] = data.get("sha256Checksum")
622        params["name_original"] = data.get("originalFilename")
623        params["properties_shared"] = data.get("properties", {})
624        params["properties_app"] = data.get("appProperties", {})
625        params["trashed"] = data.get("trashed", None)
626
627        # drive_id = data.get("driveId")
628        # """ID of the shared drive the file resides in.
629        # Only populated for items in shared drives."""
630
631        permission_ids = data.get("permissionIds")
632        # """List of permission IDs for users with access to this file."""
633
634        has_augmented_permissions = data.get("hasAugmentedPermissions")
635        # """Whether there are permissions directly on this file.
636        # This field is only populated for items in shared drives."""
637
638        permissions_list = data["fileMoverExtraPermissions"]
639
640        if included_permissions != permissions_list:
641            raise ValueError([included_permissions, permissions_list])
642        if permission_ids != [i.entry_id for i in included_permissions]:
643            raise ValueError([included_permissions, permission_ids])
644        if permission_ids != [i.entry_id for i in permissions_list]:
645            raise ValueError([permissions_list, permission_ids])
646
647        is_trashed = params.get("trashed")
648        if is_trashed is not True and is_trashed is not False:
649            raise ValueError(f"Invalid value for 'trashed': {is_trashed}.")
650
651        if has_augmented_permissions:
652            raise ValueError(data)
653
